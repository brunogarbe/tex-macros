% Preconfigured fonts

% Times New Roman
\def\setTimesNewRoman#1#2#3{
\setTextFont{XITS}{#1}
%\setMathFont{XITS}{16pt}{13pt}{10pt}
\setMathFontZero{\MathRoman}{XITS}{#1}{#2}{#3}
\setMathFontOne{\MathItalic}{XITS/I}{#1}{#2}{#3}
\setMathFontSix{\MathBold}{XITS/B}{#1}{#2}{#3}
\setMathFontFifteen{\MathBoldItalic}{XITS/BI}{#1}{#2}{#3}
\setGreekLetters{\MathItalic}
}

% Palatino
\def\setPalatino{
\setTextFont{TeX Gyre Pagella}{16pt}
%\setMathFont{TG Pagella Math}{16pt}{13pt}{10pt}
\setMathFontZero{\MathRoman}{PazoMath Nova}{16pt}{13pt}{10pt}
\setMathFontOne{\MathItalic}{PazoMath Nova/I}{16pt}{13pt}{10pt}
\setMathFontSix{\MathBold}{PazoMath Nova/B}{16pt}{13pt}{10pt}
\setMathFontFifteen{\MathBoldItalic}{PazoMath Nova/BI}{16pt}{13pt}{10pt}
\setGreekLetters{\MathItalic}
}
