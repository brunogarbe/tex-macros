% *******************************************************************
% Text Fonts

\def\setTextFont#1#2{
\font\TextFont="#1" at #2
\font\TextFontBold="#1/B" at #2
\font\TextFontItalic="#1/I" at #2
\font\TextFontBoldItalic="#1/BI" at #2
\TextFont
}

% Bold and italic
\def\rm#1{\bgroup\TextFont#1\egroup}
\def\it#1{\bgroup\TextFontItalic#1\egroup}
\def\bf#1{\bgroup\TextFontBold#1\egroup}
\def\bi#1{\bgroup\TextFontBoldItalic#1\egroup}
